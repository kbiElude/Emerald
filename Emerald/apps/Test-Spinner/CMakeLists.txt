# The name of our project is "HELLO". CMakeLists files in this project can 
# refer to the root source directory of the project as ${HELLO_SOURCE_DIR} and 
# to the root binary directory of the project as ${HELLO_BINARY_DIR}. 
cmake_minimum_required (VERSION 2.6) 
project (Test-Spinner) 

SET(CMAKE_INCLUDE_PATH "${Test-Spinner_SOURCE_DIR}/include")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY  "${EMERALD_BINARY_DIR}/bin")

MACRO(ADD_MSVC_PRECOMPILED_HEADER PrecompiledHeader PrecompiledSource SourcesVar)
  IF(MSVC)
    GET_FILENAME_COMPONENT(PrecompiledBasename ${PrecompiledHeader} NAME_WE)
    SET(Test-SpinnerSourcesPrecompiledBinary "${Test-Spinner_BINARY_DIR}/${PrecompiledBasename}.pch")
    SET(Test-SpinnerSources ${${SourcesVar}})

    SET_SOURCE_FILES_PROPERTIES(${Test-SpinnerSourcesSources}
                                PROPERTIES COMPILE_FLAGS "/W3 /analyze /Yu\"${PrecompiledHeader}\" /Fp\"{Test-SpinnerSourcesPrecompiledBinary}\" /Fp\"${Test-SpinnerSourcesPrecompiledBinary}\" "
                                           OBJECT_DEPENDS "${Test-SpinnerSourcesPrecompiledBinary}")  
    SET_SOURCE_FILES_PROPERTIES(${Test-SpinnerSourcesPrecompiledSource}
                                PROPERTIES COMPILE_FLAGS "/analyze /Yc\"${PrecompiledHeader}\" /Fp\"${PrecompiledBinary}\" "
                                           OBJECT_OUTPUTS "${Test-SpinnerSourcesPrecompiledBinary}")

    # Add precompiled header to SourcesVar
  ENDIF(MSVC)
  LIST(APPEND ${SourcesVar} ${PrecompiledSource})
ENDMACRO(ADD_MSVC_PRECOMPILED_HEADER)

IF(MSVC)
    SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /MP")
ENDIF(MSVC)

INCLUDE_DIRECTORIES("${Test-Spinner_SOURCE_DIR}")
INCLUDE_DIRECTORIES("${Test-Spinner_SOURCE_DIR}/include")
INCLUDE_DIRECTORIES("${EMERALD_SOURCE_DIR}/include")

FILE(GLOB GeneralIncludes "${Test-Spinner_SOURCE_DIR}/include/*.h")
FILE(GLOB GeneralSources "${Test-Spinner_SOURCE_DIR}/src/*.cc")

SET(AllTest-SpinnerSources ${GeneralSources} )

ADD_MSVC_PRECOMPILED_HEADER("shared.h" "src/shared.cc" AllTest-SpinnerSources)

ADD_EXECUTABLE(Test-Spinner WIN32 ${AllTest-SpinnerSources} ${GeneralIncludes})

IF (MSVC)
    SET_TARGET_PROPERTIES(Test-Spinner PROPERTIES LINK_FLAGS "/MANIFEST:NO")
ELSE (MSVC)
    # Force a 32-bit build
    SET_TARGET_PROPERTIES(Test-Spinner PROPERTIES COMPILE_FLAGS "-m32"
                                                           LINK_FLAGS    "-m32")
ENDIF (MSVC)

SET(EXECUTABLE_OUTPUT_PATH ${EMERALD_BINARY_DIR}/bin)

SOURCE_GROUP ("General includes" FILES ${GeneralIncludes})
SOURCE_GROUP ("General sources" FILES ${GeneralSources})

TARGET_LINK_LIBRARIES(Test-Spinner EMERALD)
ADD_DEPENDENCIES(Test-Spinner EMERALD)
